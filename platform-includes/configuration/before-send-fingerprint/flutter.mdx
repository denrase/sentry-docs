FutureOr<SentryEvent?> beforeSend(SentryEvent event, Hint hint) async {
  if (event.throwable is DatabaseException) {
    event = event.copyWith(fingerprint: ['database-connection-error']);
  }
  return event;
}

await Sentry.init(
  (options) {
    options.beforeSend = beforeSend;
  },
);
